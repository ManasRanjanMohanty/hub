package naren.sourcecodes;import java.math.BigDecimal;public class PassByValueExample {	public static void main(String[] args) {				// Mutable Object Example				Book book = new Book();		book.setBookName("Java");		book.setBookPrice(new BigDecimal(300.70));		System.out.println(book.getBookName());		book.changeName(book);		System.out.println(book.getBookName());		// Immutable Object Example				Apple apple = new Apple("Red");		System.out.println(apple.hashCode());		apple = apple.changeColour(apple);		System.out.println(apple.hashCode());	}}
